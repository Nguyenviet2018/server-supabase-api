✅ 1. Linear Search (Tìm kiếm tuần tự)

✔ Code:

#include <iostream>
using namespace std;

int linearSearch(int a[], int n, int x) {
    for (int i = 0; i < n; i++) {
        if (a[i] == x) return i; // tìm thấy
    }
    return -1; // không tìm thấy
}

int main() {
    int a[] = {5, 2, 7, 3, 9};
    int n = 5, x = 3;

    int kq = linearSearch(a, n, x);

    if (kq != -1) cout << "Tim thay " << x << " tai vi tri " << kq;
    else cout << "Khong tim thay!";
}

✅ 2. Binary Search (Tìm kiếm nhị phân)

⚠ Mảng phải sắp xếp trước.

✔ Code:

#include <iostream>
using namespace std;

int binarySearch(int a[], int n, int x) {
    int left = 0, right = n - 1;

    while (left <= right) {
        int mid = (left + right) / 2;

        if (a[mid] == x) return mid;
        else if (a[mid] < x) left = mid + 1;
        else right = mid - 1;
    }
    return -1;
}

int main() {
    int a[] = {1, 3, 4, 7, 9, 12};
    int n = 6, x = 7;

    int kq = binarySearch(a, n, x);

    if (kq != -1) cout << "Tim thay tai vi tri: " << kq;
    else cout << "Khong tim thay!";
}

✅ 3. Interpolation Search (Tìm kiếm nội suy)

✔ Code:

#include <iostream>
using namespace std;

int interpolationSearch(int a[], int n, int x) {
    int low = 0, high = n - 1;

    while (low <= high && x >= a[low] && x <= a[high]) {
        int pos = low + (double)(x - a[low]) * (high - low) / (a[high] - a[low]);

        if (a[pos] == x) return pos;
        else if (a[pos] < x) low = pos + 1;
        else high = pos - 1;
    }
    return -1;
}

int main() {
    int a[] = {10, 20, 30, 40, 50, 60};
    int n = 6, x = 40;

    int kq = interpolationSearch(a, n, x);
    if (kq != -1) cout << "Tim thay tai: " << kq;
    else cout << "Khong tim thay!";
}

✅ 4. Jump Search

⚠ Mảng phải sắp xếp.

✔ Code:

#include <iostream>
#include <cmath>
using namespace std;

int jumpSearch(int a[], int n, int x) {
    int step = sqrt(n);  
    int prev = 0;

    // nhảy block
    while (a[min(step, n) - 1] < x) {
        prev = step;
        step += sqrt(n);
        if (prev >= n) return -1;
    }

    // tìm tuyến tính trong block
    for (int i = prev; i < min(step, n); i++) {
        if (a[i] == x) return i;
    }
    return -1;
}

int main() {
    int a[] = {1, 4, 7, 9, 11, 14, 18, 20};
    int n = 8, x = 11;

    int kq = jumpSearch(a, n, x);
    if (kq != -1) cout << "Tim thay tai: " << kq;
    else cout << "Khong tim thay!";
}

✅ 5. Hash Search (Tìm kiếm bằng bảng băm đơn giản)

✔ Code (Hash table size = 10):

#include <iostream>
using namespace std;

#define SIZE 10

int hashTable[SIZE];

void insertHash(int value) {
    int index = value % SIZE;
    while (hashTable[index] != -1) {
        index = (index + 1) % SIZE;  // xử lý xung đột
    }
    hashTable[index] = value;
}

int searchHash(int value) {
    int index = value % SIZE;
    int start = index;

    while (hashTable[index] != -1) {
        if (hashTable[index] == value) return index;
        index = (index + 1) % SIZE;
        if (index == start) break;
    }
    return -1;
}

int main() {
    // khởi tạo bảng băm
    for (int i = 0; i < SIZE; i++) hashTable[i] = -1;

    int a[] = {5, 9, 2, 8};
    for (int v : a) insertHash(v);

    int x = 8;
    int kq = searchHash(x);
    if (kq != -1) cout << "Tim thay " << x << " tai hash index " << kq;
    else cout << "Khong tim thay!";
}
