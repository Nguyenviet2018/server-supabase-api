Dưới đây là code C++ thuật toán Kruskal chuẩn, dễ hiểu, dùng Union-Find (DSU) để tìm Minimum Spanning Tree (MST).

---

✅ C++ – Kruskal Algorithm (Full Code)

#include <bits/stdc++.h>
using namespace std;

struct Edge {
    int u, v, w;
};

vector<int> parent, sz;

// Tìm gốc của một node (DSU)
int findSet(int v) {
    if (v == parent[v]) return v;
    return parent[v] = findSet(parent[v]); // path compression
}

// Hợp nhất hai tập
bool unionSet(int a, int b) {
    a = findSet(a);
    b = findSet(b);
    if (a == b) return false; // đã cùng tập -> tạo chu kỳ
    if (sz[a] < sz[b]) swap(a, b);
    parent[b] = a;
    sz[a] += sz[b];
    return true;
}

int main() {
    int n, m;
    cin >> n >> m; // n: số đỉnh, m: số cạnh

    vector<Edge> edges(m);

    for (int i = 0; i < m; i++) {
        cin >> edges[i].u >> edges[i].v >> edges[i].w;
    }

    // Khởi tạo DSU
    parent.resize(n+1);
    sz.resize(n+1, 1);
    for (int i = 1; i <= n; i++) parent[i] = i;

    // Sắp xếp cạnh theo trọng số tăng dần
    sort(edges.begin(), edges.end(), 
        [](Edge a, Edge b) { return a.w < b.w; }
    );

    int mst_weight = 0;
    vector<Edge> mst;

    // Duyệt qua các cạnh
    for (Edge e : edges) {
        if (unionSet(e.u, e.v)) { 
            mst.push_back(e);
            mst_weight += e.w;
        }
    }

    // In kết quả
    cout << "MST Weight = " << mst_weight << "\n";
    cout << "Edges in MST:\n";
    for (auto &e : mst) {
        cout << e.u << " - " << e.v << " : " << e.w << "\n";
    }

    return 0;
}
