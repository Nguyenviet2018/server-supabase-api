#include <stdio.h>

// Hàm thực hiện tìm kiếm tuyến tính
int linearSearch(int arr[], int n, int x) {
    // Duyệt qua từng phần tử của mảng
    for (int i = 0; i < n; i++) {
        // Nếu tìm thấy phần tử bằng x, trả về chỉ số (vị trí) của nó
        if (arr[i] == x) {
            return i;
        }
    }
    // Nếu duyệt hết mảng mà không tìm thấy, trả về -1
    return -1;
}

int main() {
    int arr[] = {2, 4, 0, 1, 9};
    int n = sizeof(arr) / sizeof(arr[0]); // Tính số lượng phần tử trong mảng
    int x = 1; // Phần tử cần tìm

    int result = linearSearch(arr, n, x);

    if (result != -1) {
        printf("Phan tu %d duoc tim thay tai chi so: %d\n", x, result);
    } else {
        printf("Khong tim thay phan tu %d trong mang.\n", x);
    }

    return 0;
}
